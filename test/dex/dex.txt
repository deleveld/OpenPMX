$PROB   DEX model
$INPUT <fromdata>
$DATA "data.csv" IGNORE=@
$SUBROUTINES ADVAN11 TRANS4
$PK
    VSIZ=(WGT/70)
    CSIZ=(WGT/70)**0.75
	SES1=(2-SESS)
	SES2=(SESS-1)
    V1 =EXP(THETA(1)+ETA(1) + SES1*ETA(2) + SES2*ETA(3)) * VSIZ
    V2 =EXP(THETA(2)       ) * VSIZ
    V3 =EXP(THETA(3)+ETA(4)) * VSIZ
    CL =EXP(THETA(4)+ETA(5)) * CSIZ
    RV2=EXP(THETA(2))
    Q2 =EXP(THETA(5)) * ((V2/RV2)**0.75)
    RV3=EXP(THETA(3))
    Q3 =EXP(THETA(6)) * ((V3/RV3)**0.75)
    S1 =V1
$THETA
	( -2, 6.046010e-01, 3)
	( 1, 3.326570e+00, 5)
	( 3, 3.997030e+00, 7)
	( -2, -3.641990e-01, 1)
	( -2, 1.178030e+00, 4)
	( -2, -3.720650e-01, 2)
$OMEGA 0.1
$OMEGA BLOCK(1) 0.1
$OMEGA BLOCK SAME
$OMEGA 0.1 0.1
$SIGMA
    (0.1)
$ERROR
    IPRED=A(1)/V1
    Y=IPRED + IPRED*ERR(1)
$ESTM SIG=6 MAX=5000    METHOD=1 INTERACT SORT NOABORT POSTHOC PRINT=1
$TABLE ID TIME DVTY EVID IPRED DV PRED RES CWRES
     NOPRINT ONEHEADER NOAPPEND WRESCHOL FILE="dex.pred.txt"
$TABLE ID  V1 V2 V3 CL Q2 Q3 AGE WGT HGT M1F2
     NOPRINT ONEHEADER NOAPPEND FIRSTONLY FILE="dex.posthoc.txt"
